45. Solution


class Solution:
    def jump(self, nums: List[int]) -> int:
        # greedy, take the farthest jump
        # when you get to the end of the previous jump, increment jump and repeat
        minimum_jumps = 0
        n = len(nums)

        previous_jump_endpoint, farthest_reachable_index = 0, 0
        for i in range(n - 1):
            farthest_reachable_index = max(farthest_reachable_index, i + nums[i])

            if previous_jump_endpoint == i:
                minimum_jumps += 1
                previous_jump_endpoint = farthest_reachable_index

        return minimum_jumps
