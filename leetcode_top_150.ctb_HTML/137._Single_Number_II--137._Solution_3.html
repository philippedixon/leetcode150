137. Solution


class Solution:
    def singleNumber(self, nums: List[int]) -> int:
        unique_number = 0

        # for each bit position 
        for bit_position in range(32): 
            bit_sum = 0

            for num in nums:
                # right shift to bit_position for each number, get the bit and add it to bit_sum
                # AND 1 returns the bit in 0-th position
                bit_sum += (num >> bit_position) & 1

            # get unique bit
            unique_bit = bit_sum % 3
            # shift to bit_position in unique_number and OR to set the bit
            unique_number = unique_number | (unique_bit << bit_position)

        # if 32nd bit is used, that is the sign bit and number needs to be adjusted
        if unique_number >= (1 << 31):
            # convert to negative
            unique_number = unique_number - (1 << 32)

        return unique_number
